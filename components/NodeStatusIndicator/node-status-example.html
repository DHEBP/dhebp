How to Use in Your Project

1. Include the files:

<link rel="stylesheet" href="node-status-styles.css">
<script src="node-status-indicator.js"></script>

2. Add HTML structure:

<div class="status-bar">
    <div class="status-indicators-mini">
        <div id="connection-status" class="status-mini">
            <span class="status-dot"></span>
            <span class="status-label">NODE</span>
        </div>
    </div>
</div>

3. Initialize and use:

// Initialize
NodeStatusIndicator.init();

// Update status
NodeStatusIndicator.updateStatus('connected');
NodeStatusIndicator.updateStatus('connecting');
NodeStatusIndicator.updateStatus('disconnected');

4. Optional: Start periodic checking:

// Start periodic status check
const checkInterval = NodeStatusIndicator.startPeriodicCheck(async () => {
    // Your status check logic here
    const isOnline = await checkNodeConnection();
    return isOnline ? 'connected' : 'disconnected';
}, 5000); // Check every 5 seconds

Advanced Configuration

// Custom configuration
NodeStatusIndicator.init({
    nodeStatusSelector: '#my-node-status .dot',
    privacyStatusSelector: '#my-privacy-status .dot',
    enablePrivacyStatus: false, // Disable privacy indicator
    enableConsoleLogging: false // Disable console logs
});
